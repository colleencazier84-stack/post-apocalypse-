function drawLittleRaiderGuy(ctx, x, y, state, frameCount) {
    ctx.save();
    ctx.translate(x, y);


    // Bounce bobbing based on walking/idle frames
    let bob = 0;
    let legSwing = 0;
    if (state === 'walking' || state === 'returning') {
        bob = Math.abs(Math.sin(frameCount * 0.18)) * 3;
        legSwing = Math.sin(frameCount * 0.18) * 8;
    } else if (state === 'planting') {
        bob = Math.abs(Math.sin(frameCount * 0.3)) * 4; 
    } else if (state === 'taunting') {
        bob = Math.sin(frameCount * 0.1) * 2.5; 
    }


    // Legs
    ctx.fillStyle = '#292524';
    ctx.fillRect(-3, 6 - bob + legSwing, 2, 6);
    ctx.fillRect(1, 6 - bob - legSwing, 2, 6);


    // Spiked heavy boots
    ctx.fillStyle = '#44403c';

function isDebrisOnAnyRoad(px, py, maxDist) {
    if (isCloseToSpecificPath(px, py, path, maxDist)) return true;
    if (isCloseToSpecificPath(px, py, path2, maxDist)) return true;
    if (isCloseToSpecificPath(px, py, path3, maxDist)) return true;
    if (isCloseToSpecificPath(px, py, bossStraightPath, maxDist)) return true;
    return false;
}


function initWastelandDebris() {
    debrisList = [];
    const buildingTypes = ['shack', 'watchtower'];
    
    let attempts = 0;
    while (debrisList.length < 8 && attempts < 500) {
        let testX = Math.random() * (canvas.width - 160) + 80;
        let testY = Math.random() * (canvas.height - 160) + 80;
        
        let isCollision = isDebrisOnAnyRoad(testX, testY, 42);
        
        if (!isCollision) {
            for (let existing of debrisList) {
                let dist = Math.sqrt((existing.x - testX)**2 + (existing.y - testY)**2);
                if (dist < 85) {
                    isCollision = true;
                    break;
                }
            }
        }
        
        if (!isCollision) {
            let selectedType = buildingTypes[Math.floor(Math.random() * buildingTypes.length)];
            debrisList.push({
                x: testX,
                y: testY,
                type: selectedType,
                size: Math.random() * 8 + 24,
                color: selectedType === 'shack' ? '#542d17' : '#44403c',
                rotation: (Math.random() - 0.5) * 0.3,
                hasDish: Math.random() > 0.4,
                roofColor: Math.random() > 0.4 ? '#4a1d0d' : '#27272a',
                variationId: Math.floor(Math.random() * 3)
            });
        }
        attempts++;
    }
}


function spawnCreature() {
    if (creatures.filter(c => c.type !== 'buzzard').length >= 8) return; 
    
    const sides = ['left', 'right', 'top', 'bottom'];
    const chosenSide = sides[Math.floor(Math.random() * sides.length)];
    let cx, cy;
    
    if (chosenSide === 'left') { cx = -10; cy = Math.random() * canvas.height; }
    else if (chosenSide === 'right') { cx = canvas.width + 10; cy = Math.random() * canvas.height; }
    else if (chosenSide === 'top') { cx = Math.random() * canvas.width; cy = -10; }
    else { cx = Math.random() * canvas.width; cy = canvas.height + 10; }
    
    const types = ['coyote', 'snake', 'rabbit', 'lizard'];
    const selectedType = types[Math.floor(Math.random() * types.length)];
    
    creatures.push({
        x: cx,
        y: cy,
        speed: selectedType === 'coyote' ? 0.35 : (selectedType === 'rabbit' ? 0.65 : (selectedType === 'lizard' ? 0.40 : 0.30)),
        angle: Math.random() * Math.PI * 2,
        type: selectedType,
        legCycle: Math.random() * 100,
        changeDirTimer: Math.random() * 250 + 200,
        jumpState: 0 
    });
}

function performTowerUpgrade() {
    if (!activeSelectedTower) return;
    let lvl = activeSelectedTower.level || 1;
    let cost = lvl * 100; 
    
    if (scrap >= cost) {
        scrap -= cost;
        activeSelectedTower.level = lvl + 1;
        activeSelectedTower.range *= 1.15;
        activeSelectedTower.damage *= 1.2;
        activeSelectedTower.fireRate = Math.max(5, Math.round(activeSelectedTower.fireRate * 0.85));
        
        updateUI();
        selectTowerForUpgrade(activeSelectedTower); 
        showFloatingText(activeSelectedTower.x, activeSelectedTower.y - 25, `LVL ${lvl + 1} UPGRADED!`, "#10b981");
    } else {
        showFloatingText(activeSelectedTower.x, activeSelectedTower.y - 25, "NEED SCRAP!", "#ef4444");
    }
}

        if (dist < 25) {
            clickedTower = t;
            break;
        }
    }
    
    if (clickedTower) {
        selectTowerForUpgrade(clickedTower);
        return;
    }


    if (isCloseToPath(x, y, 28) || 
        (newPathUnlocked && isCloseToSpecificPath(x, y, path2, 28)) ||
        (newPath2Unlocked && isCloseToSpecificPath(x, y, path3, 28))) {
        showFloatingText(x, y - 10, "CANNOT PLOT ON PATH", "#ef4444");
        return;
    }


    for (let t of towers) {
        let d = Math.sqrt((t.x - x)**2 + (t.y - y)**2);
        if (d < 30) {
            showFloatingText(x, y - 10, "TOO CLOSE!", "#ef4444");
            return;
        }
    }


    activeSelectedTower = null;
    document.getElementById('upgrade-empty-state').classList.remove('hidden');
    document.getElementById('upgrade-active-state').classList.add('hidden');


    if (scrap >= selectedTowerCost) {
        const spec = towerSpecs[selectedTowerType];
        let newBunker = {
            x: x, y: y,
            type: selectedTowerType,
            range: spec.range,
            fireRate: spec.fireRate,
            damage: spec.damage,
            color: spec.color,
            barrelLen: spec.barrelLen,
            colorBarrel: spec.colorBarrel,
            bulletSize: spec.bulletSize,
            cooldown: 0,
            angle: 0,
            level: 1
        };
        towers.push(newBunker);
        scrap -= selectedTowerCost;
        updateUI();
        showFloatingText(x, y - 20, "BUNKER PLOTTED", "#34d399");
    } else {
        showFloatingText(x, y - 10, "NEED MORE SCRAP", "#f87171");
    }
});

function isCloseToPath(px, py, maxDist) {
    if (isCloseToSpecificPath(px, py, path, maxDist)) return true;
    if (newPathUnlocked && isCloseToSpecificPath(px, py, path2, maxDist)) return true;
    if (newPath2Unlocked && isCloseToSpecificPath(px, py, path3, maxDist)) return true;
    return false;
}


function isCloseToSpecificPath(px, py, targetedPath, maxDist) {
    for (let i = 0; i < targetedPath.length - 1; i++) {
        let p1 = targetedPath[i]; let p2 = targetedPath[i+1];
        let A = px - p1.x; let B = py - p1.y;
        let C = p2.x - p1.x; let D = p2.y - p1.y;
        let dot = A * C + B * D;
        let lenSq = C * C + D * D;
        let param = lenSq !== 0 ? dot / lenSq : -1;
        let xx, yy;
        if (param < 0) { xx = p1.x; yy = p1.y; }
        else if (param > 1) { xx = p2.x; yy = p2.y; }
        else { xx = p1.x + param * C; yy = p1.y + param * D; }
        let dx = px - xx; let dy = py - yy;
        if (Math.sqrt(dx * dx + dy * dy) < maxDist) return true;
    }
    return false;
}

                    cutscenePhase = 2;
                    cutsceneTimer = 0;

                    let arcY = 60;
                    let destX = 420;
                    let destY = 200;
                    
                    messengerGuy.dynamiteX = messengerGuy.x + (destX - messengerGuy.x) * messengerGuy.progress;
                    messengerGuy.dynamiteY = messengerGuy.y + (destY - messengerGuy.y) * messengerGuy.progress - Math.sin(messengerGuy.progress * Math.PI) * arcY;


                    particles.push({
                        x: messengerGuy.dynamiteX,
                        y: messengerGuy.dynamiteY,
                        vx: (Math.random() - 0.5) * 2,
                        vy: (Math.random() - 0.5) * 2,
                        life: 12,
                        size: Math.random() * 3 + 1,
                        color: '#f59e0b'
                    });


                    if (messengerGuy.progress >= 1.0) {
                        screenShake = 12;
                        for (let b = 0; b < 30; b++) {
                            particles.push({
                                x: destX,
                                y: destY,
                                vx: (Math.random() - 0.5) * 10,
                                vy: (Math.random() - 0.5) * 10,
                                life: 35,
                                size: Math.random() * 6 + 2.5,
                                color: Math.random() > 0.4 ? '#ef4444' : '#ea580c'
                            });
                        }

                        messengerGuy.state = 'returning';
                    }
                }
                else if (messengerGuy.state === 'returning') {
                    let dx = 280 - messengerGuy.x;

                    let dist = Math.sqrt(dx*dx + dy*dy);
                    if (dist < 4) {
                        messengerGuy.state = 'escaping';

                    } else {
                        messengerGuy.x += (dx/dist) * 1.5;
                        messengerGuy.y += (dy/dist) * 1.5;
                    }
                }
                else if (messengerGuy.state === 'escaping') {
                    cutsceneVehicle.x += 4.5;
                    messengerGuy.x = cutsceneVehicle.x;
                    messengerGuy.y = cutsceneVehicle.y;
                    
                    if (cutsceneVehicle.x > canvas.width + 55) {

                        cutsceneVehicle = null;
                        messengerGuy = null;

                        wave = 16;
                        updateUI();
                        document.getElementById('wave-btn').disabled = false;
                        document.getElementById('wave-btn').classList.remove('opacity-50', 'cursor-not-allowed');
                        showStatus("🚨 DEFENSES READY! COMMENCE AND DESTROY THE GOLIATH FORTRESS TANK!");
                    }
                }
            }
        }
        else if (cutsceneType === 'wave21') {
            if (cutscenePhase === 1) {
                let targetX = 350;
                let targetY = 180;
                let dx = targetX - cutsceneVehicle.x;
                let dy = targetY - cutsceneVehicle.y;
                let dist = Math.sqrt(dx*dx + dy*dy);
                if (dist < 4) {
                    cutscenePhase = 2;
                    cutsceneTimer = 0;
                } else {
                    cutsceneVehicle.x += (dx/dist) * cutsceneVehicle.speed;
                    cutsceneVehicle.y += (dy/dist) * cutsceneVehicle.speed;
                }
            }
            else if (cutscenePhase === 2) {
                bomberThrowTimer++;
                const throwIntervals = [45, 105, 165, 225];
                throwIntervals.forEach((tVal, idx) => {
                    if (bomberThrowTimer === tVal && cutsceneVehicle.targetTowers.length > idx) {
                        let targetT = cutsceneVehicle.targetTowers[idx];
                        if (towers.includes(targetT)) {
                            bullets.push({
                                startX: cutsceneVehicle.x,
                                startY: cutsceneVehicle.y,
                                x: cutsceneVehicle.x,
                                y: cutsceneVehicle.y,
                                target: targetT,
                                isDynamite: true,
                                speed: 0.015, 
                                progress: 0,
                                damage: 999, 
                                color: '#ef4444'
                            });
                            showFloatingText(cutsceneVehicle.x, cutsceneVehicle.y, "💣 BOMBS AWAY!", "#ef4444");
                        }
                    }
                });
                
                if (bomberThrowTimer > 310) {
                    cutscenePhase = 3;
                    cutsceneTimer = 0;
                }
            }
            else if (cutscenePhase === 3) {
                cutsceneVehicle.x += cutsceneVehicle.speed * 2.5;
                if (cutsceneVehicle.x > canvas.width + 40) {
                    cutsceneActive = false;
                    cutsceneVehicle = null;
                    document.getElementById('status-bar').style.opacity = 0;
                    
                    waveInProgress = false;
                    wave = 22;
                    updateUI();
                    document.getElementById('wave-btn').disabled = false;
                    document.getElementById('wave-btn').classList.remove('opacity-50', 'cursor-not-allowed');
                    showStatus("🚨 SABOTEURS IN RETREAT! READY SYSTEMS AND SEND CONVOY 22!");
                }
            }
        }
        
        for (let i = bullets.length - 1; i >= 0; i--) {
            let b = bullets[i];
            if (b.isDynamite) {
                if (!towers.includes(b.target)) {
                    bullets.splice(i, 1);
                    continue;
                }
                
                b.progress += b.speed;
                
                if (b.progress >= 1.0) {
                    let tIdx = towers.indexOf(b.target);
                    if (tIdx > -1) {
                        towers.splice(tIdx, 1);
                    }
                    for (let p = 0; p < 25; p++) {
                        particles.push({
                            x: b.target.x, y: b.target.y,
                            vx: (Math.random() - 0.5) * 8,
                            vy: (Math.random() - 0.5) * 8,

                            color: '#ea580c'

                    bullets.splice(i, 1);

                    b.x = b.startX + (b.target.x - b.startX) * b.progress;
                    b.y = b.startY + (b.target.y - b.startY) * b.progress - Math.sin(b.progress * Math.PI) * arcHeight;

        for (let i = particles.length - 1; i >= 0; i--) {
            particles[i].x += particles[i].vx;
            particles[i].y += particles[i].vy;

    // Update mid-wave bombers
    for (let i = activeMidWaveBombers.length - 1; i >= 0; i--) {
        let b = activeMidWaveBombers[i];

        for (let j = b.targetTowers.length - 1; j >= 0; j--) {
            let t = b.targetTowers[j];

                bullets.push({
                    startX: b.x,
                    startY: b.y,
                    x: b.x,
                    y: b.y,
                    target: t,
                    isDynamite: true,
                    speed: 0.02,
                    progress: 0,
                    damage: 999,
                    color: '#ef4444'

                b.targetTowers.splice(j, 1); 
                break;
            }
        }
        
        if (b.x > canvas.width + 80) {
            activeMidWaveBombers.splice(i, 1);
        }
    }


    if (waveInProgress) {
        spawnTimer++;
        if (spawnTimer > 100) { 
            spawnTimer = 0;
            if (enemiesToSpawn > 0) {
                if (wave === 16) {
                    enemies.push({
                        type: "supertank",
                        x: bossStraightPath[0].x, y: bossStraightPath[0].y,
                        targetIdx: 1,
                        speed: 0.18, 
                        hp: 12000,   
                        maxHp: 12000,
                        reward: 100,
                        size: 32,
                        color: '#1e293b',
                        name: "GOLIATH FORTRESS TANK",
                        angle: 0,
                        pathRef: bossStraightPath
                    });
                    enemiesToSpawn = 0;
                }
                else if (wave === 25) {
                    enemies.push({
                        type: "crawler_titan",
                        x: crawlerTitanPath[0].x, y: crawlerTitanPath[0].y,
                        targetIdx: 1,
                        speed: 0.22, 
                        hp: 16000,
                        maxHp: 16000,
                        reward: 150,
                        size: 36,
                        color: '#111827',
                        name: "DOOMSDAY MECH-CRAWLER",
                        angle: 0,
                        pathRef: crawlerTitanPath 
                    });
                    enemiesToSpawn = 0;
                }
                else if (wave === 30) {
                    enemies.push({
                        type: "mega_tank",
                        x: bossStraightPath[0].x, y: bossStraightPath[0].y,
                        targetIdx: 1,
                        speed: 0.12, 
                        hp: 24000,  
                        maxHp: 24000,
                        reward: 500,
                        size: 50,    
                        color: '#020617', 
                        name: "WORLD ENDER MEGA TANK",
                        angle: 0,
                        pathRef: bossStraightPath
                    });
                    enemiesToSpawn = 0;
                }
                else {
                    let rVal = Math.random();
                    let type = "pups"; 
                   
                    // Mix in huge all-terrain armor sub-bosses during Infinite Mode
                    if (isInfiniteMode && rVal > 0.85) {
                        let bossType = Math.random() > 0.5 ? "crawler_titan" : "supertank";
                        let activePath = bossType === "crawler_titan" ? crawlerTitanPath : bossStraightPath;
                        enemies.push({
                            type: bossType,
                            x: activePath[0].x, y: activePath[0].y,
                            targetIdx: 1,
                            speed: bossType === "crawler_titan" ? 0.28 : 0.22,
                            hp: (bossType === "crawler_titan" ? 12000 : 9000) * (1 + (wave - 30) * 0.25),
                            maxHp: (bossType === "crawler_titan" ? 12000 : 9000) * (1 + (wave - 30) * 0.25),
                            reward: 120,
                            size: bossType === "crawler_titan" ? 36 : 32,
                            color: bossType === "crawler_titan" ? '#111827' : '#1e293b',
                            name: bossType === "crawler_titan" ? "DOOMSDAY CRAWLER" : "GOLIATH FORTRESS",
                            angle: 0,
                            pathRef: activePath
                        });
                    } else {
                        if (rVal > 0.8 && wave > 2) {
                            type = "tanker"; 
                        } else if (rVal > 0.4 && wave > 1) {
                            type = "buggy"; 
                        }


                        let hpMultiplier = 1 + wave * 0.45; 
                        if (wave === 17 || wave === 18 || wave === 22 || wave === 23) {
                            hpMultiplier = 0.8; 
                        }
                        
                        if (wave === 1) {
                            hpMultiplier = 0.35;
                        } else if (wave === 2) {
                            hpMultiplier = 0.5;
                        }
                        
                        let activePathRef = path;
                        if (newPath2Unlocked && Math.random() > 0.6) activePathRef = path3;
                        else if (newPathUnlocked && Math.random() > 0.4) activePathRef = path2;
                       
                        if (type === "tanker") {
                            enemies.push({
                                type: "tanker",
                                x: activePathRef[0].x, y: activePathRef[0].y,
                                targetIdx: 1,
                                speed: (wave <= 2) ? 0.25 : 0.35, 
                                hp: 180 * hpMultiplier,
                                maxHp: 180 * hpMultiplier,
                                reward: 22,
                                size: 20,
                                color: '#2d3748',
                                name: "People Eater War Rig",
                                angle: 0,
                                pathRef: activePathRef
                            });
                        } else if (type === "buggy") {
                            enemies.push({
                                type: "buggy",
                                x: activePathRef[0].x, y: activePathRef[0].y,
                                targetIdx: 1,
                                speed: (wave <= 2) ? 0.45 : 0.75, 
                                hp: 75 * hpMultiplier,
                                maxHp: 75 * hpMultiplier,
                                reward: 12,
                                size: 14,
                                color: '#b45309',
                                name: "Marauder Spike Buggy",
                                angle: 0,
                                pathRef: activePathRef
                            });
                        } else {
                            enemies.push({
                                type: "pups",
                                x: activePathRef[0].x, y: activePathRef[0].y,
                                targetIdx: 1,
                                speed: (wave <= 2) ? 0.6 : 1.1, 
                                hp: 35 * hpMultiplier,
                                maxHp: 35 * hpMultiplier,
                                reward: 6,
                                size: 10,
                                color: '#4a5568',
                                name: "Scavenger War Pup",
                                angle: 0,
                                pathRef: activePathRef
                            });
                        }
                    }
                    enemiesToSpawn--;
                }
            }
        }
    }


    for (let i = enemies.length - 1; i >= 0; i--) {
        let e = enemies[i];
        if (e.hp <= 0) {
            if (e.hp > -90) {
                scrap += e.reward;
                updateUI();
                showFloatingText(e.x, e.y, `+${e.reward}⚙️`, "#fbbf24");
                
                if (e.type === 'mega_tank') {
                    triggerEndgameChoice();
                }
            }
           
            for (let d = 0; d < 12; d++) {
                particles.push({
                    x: e.x, y: e.y,
                    vx: (Math.random() - 0.5) * 6,
                    vy: (Math.random() - 0.5) * 6,
                    life: 25,
                    size: Math.random() * 3 + 1,
                    color: '#78350f'
                });
            }
            enemies.splice(i, 1);
            continue;
        }
       
        let target = e.pathRef[e.targetIdx];
        let dx = target.x - e.x;
        let dy = target.y - e.y;
        let dist = Math.sqrt(dx*dx + dy*dy);
       
        e.angle = Math.atan2(dy, dx);


        if (e.type === 'supertank' || e.type === 'mega_tank' || e.type === 'crawler_titan') {
            let crushRange = e.type === 'mega_tank' ? 65 : (e.type === 'crawler_titan' ? 55 : 45);
            for (let j = towers.length - 1; j >= 0; j--) {
                let t = towers[j];
                let d = Math.sqrt((t.x - e.x)**2 + (t.y - e.y)**2);
                if (d < crushRange) { 
                    towers.splice(j, 1);
                    showFloatingText(t.x, t.y, "💥 CRUSHED!", "#ef4444");
                    
                    for (let p = 0; p < 20; p++) {
                        particles.push({
                            x: t.x, y: t.y,
                            vx: (Math.random() - 0.5) * 5,
                            vy: (Math.random() - 0.5) * 5,
                            life: 20,
                            size: Math.random() * 4 + 1.5,
                            color: '#92400e'
                        });
                    }
                }
            }
        }


        if (dist < Math.max(2.0, e.speed)) {
            e.targetIdx++;
            if (e.targetIdx >= e.pathRef.length) {
                let structuralDamage = 1;
                if (e.type === 'buggy') structuralDamage = 2;
                if (e.type === 'tanker') structuralDamage = 4;
                if (e.type === 'supertank') structuralDamage = 10;
                if (e.type === 'crawler_titan') structuralDamage = 15;
                if (e.type === 'mega_tank') structuralDamage = 9999; 
                
                if (!isGodMode) {
                    health = Math.max(0, health - structuralDamage);
                }
                updateUI();
                enemies.splice(i, 1);
            }
        } else {
            e.x += (dx / dist) * e.speed;
            e.y += (dy / dist) * e.speed;
        }
    }


    if (enemiesToSpawn === 0 && enemies.length === 0 && waveInProgress) {
        waveInProgress = false;
        
        if (wave === 6) {
            triggerPathCutscene('wave6');
        } else if (wave === 12) {
            triggerPathCutscene('wave12');
        } else {
            if (wave === 30) {
                triggerEndgameChoice();
            } else {
                wave++;
                scrap += 30;
                updateUI();
                if (wave <= 30 || isInfiniteMode) {
                    document.getElementById('wave-btn').disabled = false;
                    document.getElementById('wave-btn').classList.remove('opacity-50', 'cursor-not-allowed');
                } else {
                    document.getElementById('wave-btn').classList.add('hidden');
                }
            }
        }
    }


    towers.forEach(t => {
        if (t.cooldown > 0) t.cooldown--;
        if (enemies.length > 0) {
            let target = null;
            let minDist = t.range;
           
            enemies.forEach(e => {
                let dist = Math.sqrt((e.x - t.x)**2 + (e.y - t.y)**2);
                if (dist < minDist) {
                    minDist = dist;
                    target = e;
                }
            });
           
            if (target) {
                t.angle = Math.atan2(target.y - t.y, target.x - t.x);
               
                if (t.cooldown === 0) {
                    bullets.push({
                        x: t.x + Math.cos(t.angle) * t.barrelLen,
                        y: t.y + Math.sin(t.angle) * t.barrelLen,
                        target: target,
                        speed: t.type === 'cannon' ? 5 : 8,
                        damage: t.damage,
                        color: t.type === 'flamer' ? '#ff4500' : '#111827',
                        type: t.type,
                        originAngle: t.angle
                    });
                    t.cooldown = t.fireRate;
                }
            }
        }
    });


    for (let i = bullets.length - 1; i >= 0; i--) {
        let b = bullets[i];
        if (b.isDynamite) continue; 
        
        if (!enemies.includes(b.target)) {
            bullets.splice(i, 1);
            continue;
        }
       
        let dx = b.target.x - b.x;
        let dy = b.target.y - b.y;
        let dist = Math.sqrt(dx*dx + dy*dy);
       
        if (dist < b.speed) {
            b.target.hp -= b.damage;
           
            if (b.type === 'cannon') {
                enemies.forEach(other => {
                    if (other !== b.target) {
                        let innerDist = Math.sqrt((other.x - b.x)**2 + (other.y - b.y)**2);
                        if (innerDist < 65) {
                            other.hp -= b.damage * 0.5; 
                        }
                    }
                });
               
                for (let s = 0; s < 18; s++) {
                    particles.push({
                        x: b.x, y: b.y,
                        vx: (Math.random() - 0.5) * 6,
                        vy: (Math.random() - 0.5) * 6,
                        life: 25,
                        size: Math.random() * 4 + 1,
                        color: '#d97706'
                    });
                }
            } else if (b.type === 'flamer') {
                particles.push({
                    x: b.x, y: b.y,
                    vx: (Math.random() - 0.5) * 2,
                    vy: (Math.random() - 0.5) * 2,
                    life: 15,
                    size: Math.random() * 3 + 2,
                    color: '#ea580c'
                });
            } else {
                particles.push({
                    x: b.x, y: b.y,
                    vx: (Math.random() - 0.5) * 3,
                    vy: (Math.random() - 0.5) * 3,
                    life: 10,
                    size: 1.5,
                    color: '#f59e0b'
                });
            }
           
            bullets.splice(i, 1);
        } else {
            b.x += (dx / dist) * b.speed;
            b.y += (dy / dist) * b.speed;
        }
    }


    for (let i = particles.length - 1; i >= 0; i--) {
        particles[i].x += particles[i].vx;
        particles[i].y += particles[i].vy;
        particles[i].life--;
        if (particles[i].life <= 0) particles.splice(i, 1);
    }
}


function updateBackground() {
    frameCount++;


    if (isPaused || nukeActive) return;


    if (frameCount % 450 === 0 && Math.random() < 0.25) {
        spawnCreature();
    }
    
    if (frameCount % 350 === 0 && Math.random() < 0.25) {
        spawnBuzzard();
    }
    
    if (frameCount % 1200 === 0 && !cutsceneActive) {
        if (Math.random() > 0.5) {
            spawnRogueRacer();
        } else {
            spawnMotorcycleGang();
        }
    }


    for (let i = 0; i < creatures.length; i++) {
        let c1 = creatures[i];
        if (c1.type === 'coyote') {
            for (let j = i + 1; j < creatures.length; j++) {
                let c2 = creatures[j];
                if (c2.type === 'coyote') {
                    let dVal = Math.sqrt((c1.x - c2.x)**2 + (c1.y - c2.y)**2);
                    if (dVal < 25) {
                        let pushAngle = Math.atan2(c1.y - c2.y, c1.x - c2.x);
                        c1.x += Math.cos(pushAngle) * 0.6;
                        c1.y += Math.sin(pushAngle) * 0.6;
                        c2.x -= Math.cos(pushAngle) * 0.6;
                        c2.y -= Math.sin(pushAngle) * 0.6;
                    }
                }
            }
        }
    }


    for (let i = creatures.length - 1; i >= 0; i--) {
        let c = creatures[i];
        
        if (c.type === 'buzzard') {
            c.circleAngle += c.circleSpeed;
            c.centerX += c.speedX;
            c.centerY += c.speedY;
            c.x = c.centerX + Math.cos(c.circleAngle) * c.circleRadius;
            c.y = c.centerY + Math.sin(c.circleAngle) * c.circleRadius;
            c.angle = c.circleAngle + Math.PI / 2; 
            
            if (c.centerX < -200 || c.centerX > canvas.width + 200 || c.centerY < -200 || c.centerY > canvas.height + 200) {
                creatures.splice(i, 1);
            }
            continue;
        }


        if (c.type === 'snake') {
            c.legCycle += 0.05; 
        } else if (c.type === 'rabbit') {
            c.legCycle += 0.04;
            c.jumpState = Math.abs(Math.sin(c.legCycle)) * 7;
        } else {
            c.legCycle += c.speed * 0.3;
        }


        c.changeDirTimer--;
        if (c.changeDirTimer <= 0) {
            c.angle = Math.random() * Math.PI * 2;
            c.changeDirTimer = Math.random() * 250 + 200;
        }


        c.x = c.x + Math.cos(c.angle) * c.speed;
        c.y = c.y + Math.sin(c.angle) * (c.type === 'rabbit' ? c.speed * 1.5 : c.speed);


        if (c.x < -60 || c.x > canvas.width + 60 || c.y < -60 || c.y > canvas.height + 60) {
            creatures.splice(i, 1);
        }
    }


    for (let i = rogueRacers.length - 1; i >= 0; i--) {
        let r = rogueRacers[i];
        
        let closestDist = 999;
        let closestObj = null;


        for (let t of towers) {
            let d = Math.sqrt((t.x - r.x)**2 + (t.y - r.y)**2);
            if (d < 80 && d < closestDist) {
                closestDist = d;
                closestObj = t;
            }
        }
        for (let db of debrisList) {
            let d = Math.sqrt((db.x - r.x)**2 + (db.y - r.y)**2);
            if (d < 80 && d < closestDist) {
                closestDist = d;
                closestObj = db;
            }
        }


        if (closestObj && closestDist < 80) {
            if (r.isReckless === undefined) {
                r.isReckless = Math.random() < 0.25; 
            }
            if (!r.isReckless) {
                let diffY = r.y - closestObj.y;
                r.vy += (diffY > 0 ? 0.04 : -0.08); 
                r.vy = Math.max(-0.6, Math.min(0.6, r.vy));
            }
        }


        let nextX = r.x + r.vx;
        let nextY = r.y + r.vy;


        if (r.type === 'ratrod') {
            r.wheelieTimer++;
            if (r.wheelieTimer > 120) { 
                r.isWheelie = !r.isWheelie;
                r.wheelieTimer = 0;
            }
            r.wheelieAngle = Math.sin(frameCount * 0.15) * 0.03 + (r.isWheelie ? -0.11 : 0); 
        }


        let deltaX = Math.abs(r.x - nextX);
        let deltaY = Math.abs(r.y - nextY);
        if (deltaX < 0.02 && deltaY < 0.02) {
            r.stuckTimer++;
        } else {
            r.stuckTimer = 0;
        }


        r.x = nextX;
        r.y = nextY;
        
        r.dustTimer++;
        if (r.dustTimer % 3 === 0) { 
            particles.push({
                x: r.x - (r.vx > 0 ? 18 : -18), 
                y: r.y + 7,
                vx: -r.vx * 0.15 + (Math.random() - 0.5) * 0.4, 
                vy: -0.05 + (Math.random() - 0.5) * 0.3,
                life: 30,
                size: Math.random() * 10 + 5,
                color: '#ca8a04' 
            });
        }


        if (r.type === 'ratrod' && frameCount % 12 === 0) {
            particles.push({
                x: r.x + (r.vx > 0 ? 4 : -4),
                y: r.y - 4,
                vx: (r.vx > 0 ? 0.5 : -0.5),
                vy: -Math.random() * 1.5,
                life: 12,
                size: Math.random() * 4 + 2,
                color: Math.random() > 0.45 ? '#ea580c' : '#fbbf24'
            });
        }


        let hasCrashed = false;
        for (let t of towers) {
            let dist = Math.sqrt((t.x - r.x)**2 + (t.y - r.y)**2);
            if (dist < 32) {
                hasCrashed = true;
                break;
            }
        }
        if (!hasCrashed) {
            for (let db of debrisList) {
                let dist = Math.sqrt((db.x - r.x)**2 + (db.y - r.y)**2);
                if (dist < 32) {
                    hasCrashed = true;
                    break;
                }
            }
        }


        if (hasCrashed || r.stuckTimer > 45) {
            for (let p = 0; p < 35; p++) {
                particles.push({
                    x: r.x, y: r.y,
                    vx: (Math.random() - 0.5) * 6,
                    vy: (Math.random() - 0.5) * 6,
                    life: Math.random() * 35 + 20,
                    size: Math.random() * 6 + 2,
                    color: Math.random() > 0.45 ? '#ea580c' : '#7c2d12'
                });
            }
            rogueRacers.splice(i, 1);
            continue;
        }


        if (r.x > canvas.width + 60 || r.y < -30 || r.y > canvas.height + 60) {
            rogueRacers.splice(i, 1);
        }
    }
}


function drawDynamicRoadSegment(ctx, baseWaypoints, currentVehicle, activeWPIndex) {
    ctx.beginPath();
    ctx.moveTo(baseWaypoints[0].x, baseWaypoints[0].y);
    
    for (let j = 1; j < 2 + activeWPIndex; j++) {
        if (baseWaypoints[j]) {
            let distToVec = Math.sqrt((baseWaypoints[j].x - currentVehicle.x)**2 + (baseWaypoints[j].y - currentVehicle.y)**2);
            if (baseWaypoints[j].x < currentVehicle.x || distToVec < 10) {
                ctx.lineTo(baseWaypoints[j].x, baseWaypoints[j].y);
            }
        }
    }
    ctx.lineTo(currentVehicle.x, currentVehicle.y);
    ctx.stroke();
}

        }
    }

function showStatus(msg) {
    const statusEl = document.getElementById('status-bar');
    if (statusEl) {
        statusEl.innerText = msg;
        statusEl.style.opacity = 1;
        
        if (msg.includes("INTERACT")) return;
        
        setTimeout(() => {
            statusEl.style.opacity = 0;
        }, 4500);
    }
}


canvas.addEventListener('click', (e) => {
    if (nukeActive && nukePhase === 6) {
        const rect = canvas.getBoundingClientRect();
        const x = ((e.clientX - rect.left) / rect.width) * canvas.width;
        const y = ((e.clientY - rect.top) / rect.height) * canvas.height;
        if (x >= canvas.width / 2 - 80 && x <= canvas.width / 2 + 80 && y >= canvas.height / 2 + 85 && y <= canvas.height / 2 + 120) {
            resetGame();
        }
    }
});


function loop() {
    if (!isPaused || nukeActive) {
        let iterations = nukeActive ? 1 : gameSpeed;
        for (let s = 0; s < iterations; s++) {
            updateGame();
        }
        updateBackground();
    }
    
    draw();
    requestAnimationFrame(loop);
}
